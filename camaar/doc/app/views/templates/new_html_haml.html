function mostrarErro(mensagem) {
  const mensagemErro = document.getElementById('mensagem-erro');
  mensagemErro.textContent = mensagem;
  mensagemErro.style.display = 'block';
  mensagemErro.scrollIntoView({ behavior: 'smooth' });
}

function validarEEnviar(event) {
  // Limpar mensagens de erro anteriores
  const mensagemErro = document.getElementById('mensagem-erro');
  mensagemErro.style.display = 'none';

  // Validação dos campos
  const nomeTemplate = document.getElementById('nome_template').value.trim();
  if (!nomeTemplate) {
    mostrarErro('Nome do template não pode ser vazio!');
    return false;
  }
  const perguntas = document.querySelectorAll('.pergunta');
  if (perguntas.length === 0) {
    mostrarErro('Adicione pelo menos uma pergunta.');
    return false;
  }
  for (let i = 0; i < perguntas.length; i++) {
    const pergunta = perguntas[i];
    const titulo = pergunta.querySelector('input[type="text"][name^="perguntas"][name$="[titulo]"]');
    if (!titulo || !titulo.value.trim()) {
      mostrarErro('O enunciado da questão é obrigatório!');
      return false;
    }
    const tipo = pergunta.querySelector('select').value;
    if (tipo === 'radio') {
      const opcoes = pergunta.querySelectorAll('input.opcao');
      // Verificar se há opções preenchidas
      let opcoesPreenchidas = 0;
      for (let j = 0; j < opcoes.length; j++) {
        if (opcoes[j].value.trim()) {
          opcoesPreenchidas++;
        }
      }
      if (opcoesPreenchidas === 0) {
        mostrarErro('A questão de marcação deve conter pelo menos uma opção de resposta!');
        return false;
      }
      // Verificar se todas as opções estão preenchidas (se houver pelo menos uma)
      for (let j = 0; j < opcoes.length; j++) {
        if (!opcoes[j].value.trim()) {
          mostrarErro(`A opção ${j + 1} da pergunta ${i + 1} não pode estar em branco.`);
          return false;
        }
      }
    }
  }
  // Serialização dos dados para envio Rails nested attributes
  const form = document.getElementById('formulario');
  const formData = new FormData(form);
  let questaoIndex = 0;
  perguntas.forEach((pergunta, i) => {
    const tipo = pergunta.querySelector('select').value;
    const titulo = pergunta.querySelector('input[type="text"][name^="perguntas"][name$="[titulo]"]').value;
    formData.append(`template[questaos_attributes][${questaoIndex}][num_questao]`, i + 1);
    formData.append(`template[questaos_attributes][${questaoIndex}][tipo]`, tipo.charAt(0).toUpperCase() + tipo.slice(1));
    formData.append(`template[questaos_attributes][${questaoIndex}][enunciado]`, titulo);
    if (tipo === 'radio') {
      const opcoes = pergunta.querySelectorAll('input.opcao');
      opcoes.forEach((opcao, j) => {
        formData.append(`template[questaos_attributes][${questaoIndex}][opcaos_attributes][${j}][num_opcao]`, j + 1);
        formData.append(`template[questaos_attributes][${questaoIndex}][opcaos_attributes][${j}][texto_opcao]`, opcao.value);
      });
    }
    questaoIndex++;
  });
  // Envio via fetch
  fetch(form.action, {
    method: 'POST',
    headers: {
      'X-CSRF-Token': document.querySelector('meta[name="csrf-token"]').content,
      'X-Requested-With': 'XMLHttpRequest'
    },
    body: formData
  }).then(response => {
    if (response.redirected) {
      window.location = response.url;
    } else {
      response.json().then(data => {
        if (data.success) {
          window.location = data.redirect_url;
        } else {
          const errorMessage = data.errors ? data.errors.join(', ') : 'Erro ao salvar template';
          mostrarErro(errorMessage);
        }
      }).catch(() => {
        response.text().then(text => mostrarErro('Erro ao salvar: ' + text));
      });
    }
  });
  event.preventDefault();
  return false;
}

function adicionarPergunta() {
  const perguntasContainer = document.getElementById("perguntas-container");
  const addQuestionRow = perguntasContainer.querySelector(".add-question-row");
  const id = contadorPerguntas++;
  const div = document.createElement("div");
  div.className = "pergunta question-card";
  div.setAttribute("data-id", id);
  div.setAttribute("data-nome", `Questão ${id + 1}`);

  div.innerHTML = `
    <label class="question-title">
      Tipo:
      <select onchange="trocarTipo(this, ${id})" class="question-select">
        <option value="texto">Texto</option>
        <option value="radio">Radio</option>
      </select>
    </label><br><br>

    <label>
      Texto:<br>
      <input type="text" name="perguntas[${id}][titulo]" placeholder="Texto" style="width: 100%;" class="question-text-input">
    </label>

    <div class="respostas" id="respostas_${id}"></div>
  `;

  // Botão de remover pergunta (exceto a primeira)
  if (contadorPerguntas > 1) {
    const btnRemove = document.createElement("button");
    btnRemove.type = "button";
    btnRemove.innerText = "X";
    btnRemove.className = "remover-pergunta";
    btnRemove.onclick = () => div.remove();
    div.prepend(btnRemove);
  }

  // Insere a pergunta antes dos botões
  perguntasContainer.insertBefore(div, addQuestionRow);
}

function trocarTipo(select, id) {
  const container = document.getElementById(`respostas_${id}`);
  container.innerHTML = "";

  if (select.value === "radio") {
    adicionarOpcao(id);
    adicionarOpcao(id);

    const botaoAdd = document.createElement("button");
    botaoAdd.type = "button";
    botaoAdd.className = "add-option-btn";
    botaoAdd.title = "Adicionar opção";
    botaoAdd.innerHTML = "+";
    botaoAdd.onclick = () => adicionarOpcao(id);

    container.appendChild(botaoAdd);
  }
}

function adicionarOpcao(id) {
  const container = document.getElementById(`respostas_${id}`);
  const opcoes = container.querySelectorAll('input.opcao');
  const input = document.createElement("input");
  input.type = "text";
  input.name = `perguntas[${id}][opcoes][]`;
  input.placeholder = "Opção de resposta";
  input.className = "opcao option-input";

  // Cria o label "Opção N"
  const label = document.createElement("label");
  label.innerText = `Opção ${opcoes.length + 1}`;
  label.appendChild(input);

  // Botão de remover opção (só aparece a partir da terceira)
  let btnRemove = null;
  if (opcoes.length >= 2) {
    btnRemove = document.createElement("button");
    btnRemove.type = "button";
    btnRemove.innerText = "X";
    btnRemove.className = "remover-opcao";
    btnRemove.onclick = () => {
      label.remove();
      btnRemove.remove();
    };
  }

  const botao = container.querySelector("button.add-option-btn");
  if (botao) {
    container.insertBefore(label, botao);
    if (btnRemove) container.insertBefore(btnRemove, botao);
  } else {
    container.appendChild(label);
    if (btnRemove) container.appendChild(btnRemove);
  }
}
